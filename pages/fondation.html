Tokens CSS

Toutes les variables CSS disponibles pour la charte.

/* ===== msyx.design tokens ===== */

:root {
  /* Surfaces */
  --primary: #0a0f1e;     /* Background principal */
  --primary-light: #0f172a; /* Background secondaire */
  --surface: rgba(30,41,59,0.6); /* Cards glass */
  --surface-solid: #1e293b; /* Cards opaques */
  --surface-light: #334155; /* Hover, toggles off */

  /* Accents */
  --accent: #3b82f6;     /* Couleur principale */
  --accent-light: #60a5fa; /* Texte accent */
  --accent-glow: rgba(59,130,246,0.12); /* Fond icones */

  /* Texte */
  --text: #f1f5f9;        /* Texte principal */
  --text-muted: #94a3b8;  /* Texte secondaire */
  --text-dim: #64748b;    /* Texte tertiaire */

  /* Gradients */
  --gradient-1: linear-gradient(135deg, #3b82f6, #8b5cf6);
  --gradient-2: linear-gradient(135deg, #06b6d4, #3b82f6);
  --gradient-3: linear-gradient(135deg, #8b5cf6, #ec4899);

  /* Bordures */
  --border: rgba(59,130,246,0.08); /* Default */
  --border-hover: rgba(59,130,246,0.25); /* Hover */

  /* Rayons */
  --radius-xs: 4px;
  --radius-sm: 8px;
  --radius: 16px;
  --radius-lg: 24px;

  /* Ombres */
  --shadow: 0 4px 30px rgba(0,0,0,0.3);
  --shadow-lg: 0 20px 60px rgba(0,0,0,0.4);

  /* Semantique */
  --success: #22c55e;
  --warning: #f59e0b;
  --danger: #ef4444;
  --info: #06b6d4;

  /* Fonts */
  /* Titres: 'Space Grotesk', sans-serif */
  /* Body: 'Inter', sans-serif */
  /* Code: 'Fira Code', monospace */
}
